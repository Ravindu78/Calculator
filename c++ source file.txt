void Calculator::on_equal_clicked()
{
    num2=ui->label->text().toDouble();
    switch (o) {
    case '+' : ans= add(num1,num2);break;
    case '-' : ans= sub(num1,num2);break;
    case '*' : ans= multi(num1,num2);break;
    case '/' : ans= div(num1,num2);break;
       // case '>' : ans= signChange(sign);break;

    }
    ui->label->setText(QString::number (ans));
}
